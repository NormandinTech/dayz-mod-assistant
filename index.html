override void OnContact(string zoneName, vector localPos,
IEntity other, Contact contact) {
super.OnContact(zoneName, localPos, other, contact);
if (contact.Impulse > 500) {
ProcessDirectDamage(0, "", zoneName,
"DamageSystemZone", localPos, contact.Impulse * 0.1);
